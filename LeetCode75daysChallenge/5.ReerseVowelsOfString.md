[Problem Statement]()

Given a string s, reverse only all the vowels in the string and return it.

The vowels are 'a', 'e', 'i', 'o', and 'u', and they can appear in both lower and upper cases, more than once.

 

Example 1:

Input: s = "IceCreAm"

Output: "AceCreIm"

Explanation:

The vowels in s are ['I', 'e', 'e', 'A']. On reversing the vowels, s becomes "AceCreIm".

Example 2:

Input: s = "leetcode"

Output: "leotcede"

solution:
bool isVowel(char c)
{
         return (c=='a' || c=='A' || c=='e' ||
            c=='E' || c=='i' || c=='I' ||
            c=='o' || c=='O' || c=='u' ||
            c=='U');
}
string reverseVowels(string s) 
{
        int i=0,j=s.length();
       while(i<j){
        if(!isVowel(s[i])){
            i++;
            continue;
        }
        if(!isVowel(s[j])){
            j--;
            continue;
        }
        char t=s[i];
        s[i]=s[j];
        s[j]=t;
        i++;
        j--;
       }
       return s;
}
